Useful commands for Qwen project:

# 1. Set PEM permission locally
chmod 400 ~/Downloads/memos.pem

# 2. SSH into ECS
ssh -i ~/Downloads/memos.pem root@47.236.145.69

# 3. Go to project folder
cd ~/qwen
# or wherever you cloned it

# 4. Edit environment file
nano .env

# Save in nano:
# Ctrl+O
# Enter
# Ctrl+X

# 5. Check current containers
docker compose ps

# 6. View logs
docker compose logs -f api
docker compose logs -f frontend
docker compose logs -f worker

# 7. Restart containers after .env change
docker compose down
docker compose up -d --build

# 8. Pull latest GitHub code into ECS
cd ~/qwen
git status
git pull origin main

# If your branch is master instead:
git pull origin master

# 9. Rebuild after pulling latest code
docker compose down
docker compose up -d --build

# 10. If frontend env changed, force rebuild frontend
docker compose build --no-cache frontend
docker compose up -d frontend

# 11. Check API health
curl http://localhost:8000/health

# 12. Check running ports
docker compose ps
ss -tulpn | grep -E '3000|8000|9090|3001'

# 13. Stop everything
docker compose down

# 14. Start again
docker compose up -d

# 15. Restart only one service
docker compose restart api
docker compose restart frontend
docker compose restart worker
